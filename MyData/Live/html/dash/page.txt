function switchChanged(name){
	var position = document.getElementById(name).checked;
	$.post("/live",
	{
		type: "switch",
		name: name,
		value: position
	},
	function(data, status){
		
	});
	
}

function buttonClicked(name){
	$.post("/live",
	{
		type: "switch",
		name: name,
		value: "btnClick!"
	},
	function(data, status){

	});
}

function update(){
	$.post("/live",
	{
		type: "get"
	},
	function(data, status){
		var info = data.split("!");
		for (i = 0; i < info.length; i++) {
			var item = info[i].split("@");

			if(item[2] == "switch"){
				if(item[1] == "true"){
					document.getElementById(item[0]).checked = true;
				}else{
					document.getElementById(item[0]).checked = false;
				}
			}else if(item[2] == "text"){
				document.getElementById(item[0]).value = item[1];
			}
		}
		setTimeout(function(){
			update();
		}, 1000);
	});
}
</script>
